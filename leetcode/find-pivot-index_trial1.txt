class Solution:
    def pivotIndex(self, nums: List[int]) -> int:
        sum_left = 0
        sum_right = sum(nums) - nums[0]
        if sum_left == sum_right:
            return 0

        for idx in range(1, len(nums)):
            sum_left += nums[idx - 1]
            sum_right -= nums[idx]
            if sum_left == sum_right:
                return idx
        return -1
