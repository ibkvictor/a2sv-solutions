import math
class Solution:
    def maxTurbulenceSize(self, arr: List[int]) -> int:
        def start_func(prev, curr):
            if prev > curr:
                return -1

            elif curr > prev:
                return 1

            else:
                return 0

        start = True
        max_len = 1
        prev = 0
        length = 1
        for curr in range(1, len(arr)):
            if start == True:
                marker = start_func(arr[prev], arr[curr])
                start = False

            if marker > 0: 
                if arr[curr] > arr[prev]:
                    marker *= -1
                    length += 1
                    prev = curr

                elif arr[prev] > arr[curr]:
                    # maker = start(arr[prev], arr[curr])
                    prev = curr
                    length = 2
                    maker = 1

                else: # equal to
                    prev = curr
                    length = 1
                    start = True

            elif marker < 0:
                if arr[curr] < arr[prev]:
                    marker *= -1
                    length += 1
                    prev = curr

                elif arr[prev] < arr[curr]:
                    # maker = start(arr[prev], arr[curr])
                    prev = curr
                    length = 2
                    marker = -1

                else: # equal to
                    prev = curr
                    length = 1
                    start = True
                
            else:
                prev = curr
                length = 1
                start = True

            max_len = max(length, max_len)
            # print(marker)
        return max_len
