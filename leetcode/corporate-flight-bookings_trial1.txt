class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        result = [0] * (n + 1) # to include the last spot for negating

        for first, last, n_seats in bookings:
            result[first - 1] += n_seats
            result[last] += -n_seats
            # print(result)

        for i in range(1, n):
            result[i] += result[i - 1]

        return result[:n]
