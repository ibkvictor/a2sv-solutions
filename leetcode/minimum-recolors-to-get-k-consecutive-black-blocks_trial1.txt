import math
class Solution:
    def minimumRecolors(self, blocks: str, k: int) -> int:
        i = 0
        min_recolors = math.inf
        count = {"W": 0, "B": 0}
        for j in range(len(blocks)):
            count[blocks[j]] += 1
            while i < j and j - i + 1 > k:
                count[blocks[i]] -= 1
                # if not count[blocks[j]]:
                #     del count[blocks[j]]
                i += 1
            # print(count, i, j)
            if j - i + 1 == k:
                min_recolors = min(min_recolors, count["W"])
                # print(min_recolors)

        return min_recolors
