import math
class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        closest = math.inf
        nums.sort()

        for i in range(len(nums)-2):
            j = i + 1
            k = len(nums) - 1
            while j < k:
                sum_value = nums[i] + nums[j] + nums[k]
                if sum_value > target:
                    if abs(sum_value - target) < abs(closest - target):
                        closest = sum_value
                    k -= 1

                elif sum_value < target:
                    if abs(sum_value - target) < abs(closest - target):
                        closest = sum_value
                    j += 1

                else:
                    # if abs(sum_value - target) < abs(closest - target):
                    closest = sum_value
                    # j += 1
                    # k -= 1
                    break

                # # if abs(sum_value - target) > abs(closest - target):
                # #     if sum_value - target > 0:
                # #         k -= 1


                    
                # # else:
                # #     closest = sum_value
                # #     j += 1
                #     k -= 1

        return closest
