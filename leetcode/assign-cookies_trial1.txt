class Solution:
    def findContentChildren(self, g: List[int], s: List[int]) -> int:
        g.sort(reverse = True)
        s.sort(reverse = True)

        count = 0
        i = 0
        j = 0
        while i < len(s) and j < len(g):
            if s[i] >= g[j]:
                count += 1
                i += 1
                j += 1

            else:
                j += 1

        return count
