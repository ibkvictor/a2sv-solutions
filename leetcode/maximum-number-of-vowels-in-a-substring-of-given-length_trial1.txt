from collections import defaultdict
class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        i = 0
        count_vowels = 0
        max_sum = 0
        set_vowels = {"a", "e", "i", "o", "u"}
        for j in range(len(s)):

            while i < j and j - i + 1 > k:
                if s[i] in set_vowels:
                    count_vowels -= 1
                i += 1

            if s[j] in set_vowels:
                count_vowels += 1

            max_sum = max(count_vowels, max_sum)

        return max_sum
